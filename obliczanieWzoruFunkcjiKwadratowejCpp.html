#include <iostream>

using namespace std;


float calculatingthematrixdeterminant3x3(float a1, float a2, float a3, float b1, float b2, float b3, float c1, float c2, float c3)
{
float w;
w = a1*b2*c3 + a2*b3*c1 + a3*b1*c2 - a3*b2*c1 - a2*b1*c3 - a1*b3*c2;
return w;
}



int main()
{
float x1, y1, z1, w1, x2, y2, z2, w2, x3, y3, z3, w3, W, Wx, Wy, Wz, x, y, z, x1Tymczasowy, x2Tymczasowy, x3Tymczasowy;

cout << "Współrzędne wprowadzaj oddzielone spacją\n";

//cout << "Wzór funkcji kwadratowej: f(x) = ax² + bx + c\n";

cout << "\nWprowadź współrzędne 1 punktu: ";
cin >> x1;
cin >> y1;
cout << "\nWprowadź współrzędne 2 punktu: ";
cin >> x2;
cin >> y2;
cout << "\nWprowadź współrzędne 3 punktu: ";
cin >> x3;
cin >> y3;

w1 = y1;
w2 = y2;
w3 = y3;

x1Tymczasowy = x1;
x2Tymczasowy = x2;
x3Tymczasowy = x3;

x1 = x1Tymczasowy * x1Tymczasowy;
y1 = x1Tymczasowy;
z1 = 1;


x2 = x2Tymczasowy * x2Tymczasowy;

x3 = x3Tymczasowy * x3Tymczasowy;

W = calculatingthematrixdeterminant3x3(x1, y1, z1, x2, y2, z2, x3, y3, z3);
Wx = calculatingthematrixdeterminant3x3(w1, y1, z1, w2, y2, z2, w3, y3, z3);
Wy = calculatingthematrixdeterminant3x3(x1, w1, z1, x2, w2, z2, x3, w3, z3);
Wz = calculatingthematrixdeterminant3x3(x1, y1, w1, x2, y2, w2, x3, y3, w3);

x = Wx / W;
