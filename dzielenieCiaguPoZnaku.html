#include <iostream>

using namespace std;


string slowa[9999];

int tokenize(string s, string del)
{
int licznik = 0;

int start = 0;
int end = s.find(del);
while (end != -1) {

slowa[licznik] = s.substr(start, end - start);
licznik++;

start = end + del.size();
end = s.find(del, start);
}

slowa[licznik] = s.substr(start, end - start);
return licznik;
}

{
