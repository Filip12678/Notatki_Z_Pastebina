std::string ReplaceAll(std::string str, const std::string& from, const std::string& to) {
size_t start_pos = 0;
while((start_pos = str.find(from, start_pos)) != std::string::npos) {
str.replace(start_pos, from.length(), to);
start_pos += to.length(); // Handles case where 'to' is a substring of 'from'
}
return str;
}

string rozkladLiczbNaCzynnikiPierwsze(unsigned long long n)
{

long double pom;
unsigned long long k, pierw;

pierw = sqrt(n);
k = 2;
string liczbyPierwsze = "";

while (n > 1 && k <= pierw)
{

while (n % k == 0)
{
liczbyPierwsze = liczbyPierwsze + to_string(k) + " ";
n /= k;
}
++k;


}


if (n > 1)
{
liczbyPierwsze = liczbyPierwsze + to_string(n) + " ";
}

return liczbyPierwsze;

}
