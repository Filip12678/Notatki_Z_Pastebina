double det(double A[5][5], int N)
{

double c, r=1;
for(int i = 0; i < N; i++) {
for(int k = i+1; k < N; k++) {
c = A[k][i] / A[i][i];
for(int j = i; j < N; j++)
A[k][j]= A[k][j] - c*A[i][j];
}
}
for (int i = 0; i < N; i++)
r *=A[i][i];
return r;
}
