bool isLeap(int y)
{
if (y%100 != 0 && y%4 == 0 || y %400 == 0)
return true;

return false;
}

int offsetDays(int d, int m, int y)
{
int offset = d;

switch (m - 1)
{
case 11:
offset += 30;
case 10:
offset += 31;
case 9:
offset += 30;
case 8:
offset += 31;
case 7:
offset += 31;
case 6:
offset += 30;
case 5:
offset += 31;
case 4:
offset += 30;
case 3:
offset += 31;
case 2:
offset += 28;
case 1:
offset += 31;
}

if (isLeap(y) && m > 2)
offset += 1;

return offset;
}

void revoffsetDays(int offset, int y, int *d, int *m)
{
int month[13] = { 0, 31, 28, 31, 30, 31, 30,
31, 31, 30, 31, 30, 31 };

if (isLeap(y))
month[2] = 29;

int i;
for (i = 1; i <= 12; i++)
{
if (offset <= month[i])
break;
offset = offset - month[i];
}

*d = offset;
*m = i;
}

int dodawanieDniWynikDzien(int d1, int m1, int y1, int x)
{
int offset1 = offsetDays(d1, m1, y1);
int remDays = isLeap(y1)?(366-offset1):(365-offset1);

int y2, offset2;
if (x <= remDays)
{
y2 = y1;
offset2 = offset1 + x;
}

else
{
x -= remDays;
y2 = y1 + 1;
int y2days = isLeap(y2)?366:365;
while (x >= y2days)
{
x -= y2days;
y2++;
y2days = isLeap(y2)?366:365;
}
offset2 = x;
}

int m2, d2;
revoffsetDays(offset2, y2, &d2, &m2);

return d2;
}

string weekday[7] = {"6","7","1","2", "3","4","5"};
string zellersAlgorithm(int day, int month, int year){
int mon;
if(month > 2)
mon = month;
else{
mon = (12+month);
year--;
}
int y = year % 100;
int c = year / 100;
int w = (day + floor((13*(mon+1))/5) + y + floor(y/4) + floor(c/4) + (5*c));
w = w % 7;
return weekday[w];
}


int rdn(int y, int m, int d) {
if (m < 3)
y--, m += 12;
return 365*y + y/4 - y/100 + y/400 + (153*m - 457)/5 + d - 306;
}


int obliczanieIlosciPowtorzenDanegoDniaPomiedzyDatami(int dzien1, int miesiac1, int rok1, int dzien2, int miesiac2, int rok2, string dzienTygodniaDoObliczenia)
{


float roznicaDni = rdn(rok2, miesiac2, dzien2) - rdn(rok1, miesiac1, dzien1);
int iloscWystapienDnia = 0;
string dzienTygodnia;

int dzienTymczasowy = 0;
int miesiacTymczasowy = 0;
int rokTymczasowy = 0;

for (int i = 0; i < roznicaDni; i++)
{

dzienTygodnia = zellersAlgorithm(dzien1, miesiac1, rok1);



if (dzienTygodnia == dzienTygodniaDoObliczenia)
{
iloscWystapienDnia = iloscWystapienDnia + 1;

}

dzienTymczasowy = dzien1;

dzien1 = dodawanieDniWynikDzien(dzienTymczasowy, miesiacTymczasowy, rokTymczasowy, 1);


}

return iloscWystapienDnia;

}
